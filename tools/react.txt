5.点击每一项div的时候，删除？

 onClick={this.handleItemClick.bind(this,index)}

 //点击每一项的时候，进行删除
  handleItemClick(index){
    //console.log(index);  扩展运算符
    const list = [...this.state.list];
    list.splice(index,1);   
    this.setState({
      list:list
    });
  }
